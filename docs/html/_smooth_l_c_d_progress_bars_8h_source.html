243 void makeCharMask(uint8_t out[barstyle::CharPatRows], const uint8_t xMask[2][barstyle::CharPatRows], uint8_t mask) const
244 {
245 if (styleOrientation() == barstyle::Vertical)
246 {
247 for (byte i = 0; i < barstyle::CharPatRows; i++, mask >>= 1)
248 {
249 out[i] = (mask & 1) ? (xMask[0][i] | xMask[1][i]) : xMask[1][i];
250 }
251 }
252 else // Horizontal
253 {
254 for (byte i = 0; i < barstyle::CharPatRows; i++)
255 {
256 out[i] = (mask & xMask[0][i]) | xMask[1][i];
257 }
258 }

302 for (byte i = 0; i < barstyle::CharPatRows; i++)
303 {

448 void showProg(int val, byte n)
449 {
450 const barstyle::bspar& barStylePar = barDisp.stylePar();
451 const byte cellDim = barStylePar.dir ? barstyle::CharPatRows : barstyle::CharPatCols;
452 const byte rem = (cellDim - (val % cellDim));
453 uint8_t mask = barStylePar.dir ?
454 (barstyle::AllRows << rem) :
455 ((barstyle::AllCols << rem));
456 byte full = min(max(0, val / cellDim - 1), par.len - 2);
457 const byte partial = (val > cellDim && val < (par.len - 1) * cellDim);
458 byte blank = max(0, par.len - full - 2 - partial);
459
460 // determine and prepare the user defined characters
461 if (val < cellDim) // still filling the first block, create and use the middle char
462 {
463 barDisp.makeChar(DISP::ProgBarCharPartial + n, barDisp.getStyle().startMask, mask);
464 }
465 if (partial)
466 {
467 barDisp.makeChar(DISP::ProgBarCharPartial + n, barDisp.getStyle().middleMask, mask);
468 }
469 if (val > (par.len - 1) * cellDim) // last block partially (or fully) filled, create and use the middle char
470 {
471 if (val >= par.len * cellDim) // special case val == size() - last block completely full
472 {
473 mask = (barStylePar.dir ? barstyle::AllRows : barstyle::AllCols);
474 }
475 barDisp.makeChar(DISP::ProgBarCharPartial + n, barDisp.getStyle().endMask, mask);
476 }
477
478 barDisp.dsplay().setCursor(par.col, par.row);
479 barDisp.dsplay().write((val < cellDim) ? (DISP::ProgBarCharPartial + n) : DISP::ProgBarCharBegin);
480
481 // show on the display
482 byte crow = par.row;
483 while (full--)
484 {
485 if (barStylePar.dir)
486 {
487 barDisp.dsplay().setCursor(par.col, --crow);
488 }
489 barDisp.dsplay().write(DISP::ProgBarCharFull);
490 }
491 if (partial)
492 {
493 if (barStylePar.dir)
494 {
495 barDisp.dsplay().setCursor(par.col, --crow);
496 }
497 barDisp.dsplay().write(DISP::ProgBarCharPartial + n);
498 }
499 while (blank--)
500 {
501 if (barStylePar.dir)
502 {
503 barDisp.dsplay().setCursor(par.col, --crow);
504 }
505 barDisp.dsplay().write(DISP::ProgBarCharBlank);
506 }
507 if (barStylePar.dir)
508 {
509 barDisp.dsplay().setCursor(par.col, --crow);
510 }
511 barDisp.dsplay().write((val > (par.len - 1) * cellDim) ? (DISP::ProgBarCharPartial + n) : DISP::ProgBarCharEnd);
512 }
513
514 //data
515 const DISP& barDisp; // reference to the display wrapper object
